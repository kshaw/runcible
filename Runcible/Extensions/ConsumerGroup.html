def self.generate_content(type_id, units)
  content = []
  units.each do |unit|
    content_unit = {}
    content_unit[:type_id] = type_id
    content_unit[:unit_key] = { :name => unit }
    content.push(content_unit)
  end
  content
end

def self.make_consumer_criteria(consumer_ids)
  {:criteria =>
        {:filters =>
          {:id =>{"$in" =>consumer_ids}}
        }
  }
end
